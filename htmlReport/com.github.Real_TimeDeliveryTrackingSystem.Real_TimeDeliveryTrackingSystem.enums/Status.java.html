import com.fasterxml.jackson.annotation.JsonProperty;

    MAINTENANCE("MAINTENANCE"),
    @JsonProperty("out_of_service")
    OUT_OF_SERVICE("OUT_OF_SERVICE");

    Status(String status) {
        this.status = status;
    }

    public String getStatus() {
